<script>  
    // JAVASCRIPT LOGIC  
    document.addEventListener('DOMContentLoaded', () => {  
        const animatedText = document.getElementById('animated-text');  
        const backgroundContainer = document.getElementById('background-words');  
          
        // Slang list  
        const slangWords = ['PAGAL', 'BENDI', 'HUGLI', 'GELCHAPI', 'CHACUNDARI'];   
        const totalWordsToGenerate = 150;   

        /** * Dynamically generates and places the background words randomly.  
         */  
        function generateBackground() {  
            for (let i = 0; i < totalWordsToGenerate; i++) {  
                const word = document.createElement('span');  
                  
                word.textContent = slangWords[i % slangWords.length];  
                word.className = 'bg-slang';  

                // Random placement and rotation  
                word.style.top = `${Math.random() * 100}vh`;  
                word.style.left = `${Math.random() * 100}vw`;  
                  
                const rotation = Math.floor(Math.random() * 180) - 90;   
                word.style.transform = `rotate(${rotation}deg)`;  
                  
                // Stagger the animation start for a more dynamic pulse  
                word.style.animationDelay = `${Math.random() * 5}s`;   
                  
                backgroundContainer.appendChild(word);  
            }  
        }  

        /** * Starts the main central animation.  
         */  
        function startAnimation() {  
            animatedText.classList.remove('hidden');  
            animatedText.classList.add('animate');  
        }  

        /** * Resets the main animation and schedules the next loop.  
         */  
        function resetAndLoop() {  
            animatedText.classList.remove('animate');  
            animatedText.classList.add('hidden');  
              
            // Restart the cycle after a minimal delay  
            setTimeout(startAnimation, 50);   
        }  

        // Set up the listener to automatically loop when the main animation finishes  
        animatedText.addEventListener('animationend', resetAndLoop);  

        // Initial calls  
        generateBackground();   
        startAnimation();   
    });  
</script>
